        \ket{\Psi^+} &= \frac{1}{\sqrt{2}}\left[ \ket{10} + \ket{01}\right] = \frac{1}{\sqrt{2}}\left[\begin{bmatrix} 0 \\ 1 \\ 0 \\ 0 \end{bmatrix} + \begin{bmatrix} 0 \\ 0 \\ 1 \\ 0 \end{bmatrix}\right] = \frac{1}{\sqrt{2}}\begin{bmatrix} 0 \\ 1 \\ 1 \\ 0\end{bmatrix}\\
        \ket{\Psi^-} &= \frac{1}{\sqrt{2}}\left[ \ket{10} - \ket{01}\right] = \frac{1}{\sqrt{2}}\left[\begin{bmatrix} 0 \\ 1 \\ 0 \\ 0 \end{bmatrix} - \begin{bmatrix} 0 \\ 0 \\ 1 \\ 0 \end{bmatrix}\right] = \frac{1}{\sqrt{2}}\begin{bmatrix} 0 \\ 1 \\ -1 \\ 0\end{bmatrix}\\
    \end{split}
\end{align*}
