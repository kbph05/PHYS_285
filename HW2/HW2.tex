\section*{Question 1a}
\textbf{Probabilities in z-direction (up and down):} \\
Possible outcomes: either in positive or negative z-direction
\begin{equation}
    \begin{split}
        P_{+z} &= |\braket{ \uparrow | \Psi}|^{2} \\
        &= \left| \frac{1}{\sqrt{13}}(3\braket{\uparrow | \uparrow} + 2j\braket{\uparrow | \downarrow})\right|^2 \\
        &= \frac{1}{13}\left|(3\braket{\uparrow | \uparrow} + 2j\braket{\uparrow | \downarrow})^{2}\right| \\
        &= \frac{1}{13}\left|(3 + 0)^{2}\right| \\
        &= \frac{1}{13}\left(\sqrt{9 + 0}\right)^{2} = \frac{9}{13} \approx 69.23\% \\
    \end{split}
\end{equation}


\begin{equation}
    \begin{split}
        P_{-z} &= |\braket{ \downarrow | \Psi}|^{2} \\
        &= \left|\frac{1}{\sqrt{13}}(3\braket{\downarrow | \uparrow} - 2j\braket{\downarrow | \downarrow})\right|^2 \\
        &= \frac{1}{13}\left|(3\braket{\downarrow | \uparrow} - 2j\braket{\downarrow | \downarrow})^{2}\right| \\ 
        &= \frac{1}{13}\left|(0 - 2j)^{2}\right| \\ 
        &= \frac{1}{13}\left(\sqrt{0 + 4}\right)^{2} = \frac{4}{13} \approx 30.77\% \\
    \end{split}
\end{equation}
\[
\text{Or: } P_{-z} = 100 - 69.23 = 30.77\%
\]

\textbf{Probabilities in the x-direction (left and right):} \\
Possible Outcomes: Since the original state is either in the positive z or negative z direction initially, a measurement in a different axes mixes the state. The probability of entering through the positive or negative x direction will be equal.

\begin{equation}
    \begin{split}
        P_{+x} &= \left|\braket{ \rightarrow | \Psi}\right|^{2} \\
        &= \frac{1}{13}\left|\left(3\braket{\rightarrow | \uparrow} + 2i\braket{\rightarrow | \downarrow}\right)^{2}\right| \\
        &= \frac{1}{13}\left|\left(3\left(\frac{1}{\sqrt{2}}\bra{\uparrow} + \frac{1}{\sqrt{2}}\bra{\downarrow}\right)\ket{\uparrow} + 2i\left(\frac{1}{\sqrt{2}}\bra{\uparrow} + \frac{1}{\sqrt{2}}\bra{\downarrow}\right)\ket{\downarrow}\right)\right|^2 \\
        &= \frac{1}{13}\left|\left(\frac{3}{\sqrt{2}} + \frac{2i}{\sqrt{2}}\right)^2\right| \\
        &= \frac{1}{26}\left|\left(3 + 2i\right)^2\right| \\
        &= \frac{1}{26}(\sqrt{9 + 4})^2  = \frac{\sqrt{169}}{26} = \frac{13}{26} = 50\% \\
    \end{split}
\end{equation}

\begin{equation}
    \begin{split}
        P_{-x} &= \left|\braket{ \leftarrow | \Psi}\right|^{2} \\
        &= \frac{1}{13}\left|\left(3\braket{\leftarrow | \uparrow} + 2i\braket{\leftarrow | \downarrow}\right)^{2}\right| \\
        &= \frac{1}{13}\left|\left(3\left(\frac{1}{\sqrt{2}}\bra{\uparrow} - \frac{1}{\sqrt{2}}\bra{\downarrow}\right)\ket{\uparrow} + 2i\left(\frac{1}{\sqrt{2}}\bra{\uparrow} - \frac{1}{\sqrt{2}}\bra{\downarrow}\right)\ket{\downarrow}\right)\right|^2 \\
        &= \frac{1}{13}\left|\left(\frac{3}{\sqrt{2}} - \frac{2i}{\sqrt{2}}\right)^2\right| \\
        &= \frac{1}{26}\left|\left(3 - 2i\right)^2\right| \\
        &= \frac{1}{26}(\sqrt{9 + 4})^2 = \frac{13}{26} = 50\% \\
    \end{split}
\end{equation}

\section*{Question 2}
We first know:
\begin{equation}
    \ket{-}_{y} = \frac{1}{\sqrt2}\ket{\uparrow} + \frac{1}{\sqrt2}\ket{\downarrow}
\end{equation}
\begin{equation}
    \ket{+}_{y} = \frac{1}{\sqrt2}\ket{\uparrow} - \frac{1}{\sqrt2}\ket{\downarrow}    
\end{equation}
\begin{equation}
    P_{-y} = |\braket{-|\Psi}|^2    
\end{equation}
\begin{equation}
    P_{+y} = |\braket{+|\Psi}|^2 
\end{equation} \\

1. Solve for the expectation value using an arbituary vector $\Psi$ :
\begin{equation}
    \begin{split}
        \braket{S_y} &= P_{+y}\left(\frac{\hbar}{2}\right)  + P_{-y}\left(-\frac{\hbar}{2}\right) \\
        &= \frac{\hbar}{2}\left( |\braket{+|\Psi}|^2 - |\braket{-|\Psi}|^2\right) \\
        &= \frac{\hbar}{2}\left( \braket{\Psi|+}\braket{+|\Psi} - \braket{\Psi|-}\braket{-|\Psi}\right) \\
        &= \bra{\Psi}\left(\frac{\hbar}{2}\ket{+}\bra{+} - \frac{\hbar}{2}\ket{-}\bra{-}\right)\ket{\Psi} \\
    \end{split}
\end{equation}

2. Define each ket-bra vectors in terms of their basis (z-direction)
\begin{equation}
    \begin{split}
        \ket{-}\bra{-} &= \left(\frac{1}{\sqrt{2}}\bra{\uparrow} - \frac{i}{\sqrt{2}}\bra{\downarrow}\right)\left(\frac{1}{\sqrt{2}}\ket{\uparrow} - \frac{i}{\sqrt{2}}\ket{\downarrow}\right) \\
        &= \frac{1}{\sqrt{2}}\frac{1}{\sqrt{2}}\left(\begin{bmatrix} 1 \\ 0 \end{bmatrix} - \begin{bmatrix} 0 \\ i \end{bmatrix}\right)\left(\begin{bmatrix} 1 & 0 \end{bmatrix} + \begin{bmatrix} 0 & i \end{bmatrix}\right) \\
        &= \frac{1}{2}\begin{bmatrix} 1 \\ -i \end{bmatrix}\begin{bmatrix} 1 & i \end{bmatrix} \\
        &= \frac{1}{2}\begin{bmatrix} 1 & i \\ -i & -i^2 \end{bmatrix} \\
    \end{split}
\end{equation}
\begin{equation}
    \begin{split}
        \ket{+}\bra{+} &= \left(\frac{1}{\sqrt{2}}\bra{\uparrow} + \frac{i}{\sqrt{2}}\bra{\downarrow}\right)\left(\frac{1}{\sqrt{2}}\ket{\uparrow} + \frac{i}{\sqrt{2}}\ket{\downarrow}\right) \\
        &= \frac{1}{\sqrt{2}}\frac{1}{\sqrt{2}}\left(\begin{bmatrix} 1 \\ 0 \end{bmatrix} + \begin{bmatrix} 0 \\ i \end{bmatrix}\right)\left(\begin{bmatrix} 1 & 0 \end{bmatrix} - \begin{bmatrix} 0 & i \end{bmatrix}\right) \\
        &= \frac{1}{2}\begin{bmatrix} 1 \\ i \end{bmatrix}\begin{bmatrix} 1 & -i \end{bmatrix} \\
        &= \frac{1}{2}\begin{bmatrix} 1 & -i \\ i & -i^2 \end{bmatrix} \\
    \end{split}
\end{equation}
3. Plug each matrix back into the expectation value equation (1)
\begin{equation}
    \begin{split}
        \braket{S_y} &= \frac{\hbar}{2}\left(\frac{1}{2}\begin{bmatrix} 1 & -i \\ i & -i^2 \end{bmatrix} - \frac{1}{2}\begin{bmatrix} 1 & i \\ -i & -i^2 \end{bmatrix}\right) \\
        &= \frac{1}{2}\frac{\hbar}{2}\begin{bmatrix} 0 & -2i \\ 2i & 0 \end{bmatrix} \\
        &= \frac{\hbar}{2}\begin{bmatrix} 0 & -i \\ i & 0 \end{bmatrix} \\
    \end{split}
\end{equation}
